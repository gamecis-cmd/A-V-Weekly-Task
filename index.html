    <script>

        const WEB_APP_URL = "https://script.google.com/macros/s/AKfycbyJikQio8vML4OvkchM16cqNz0TtHszQxXfwUixGcFfEHq76iELOpawgnDMr0kj1OXk/exec";

        let state = { tasks: [], currentDay: 'Monday' };


        function updateDateHeader() {

            const now = new Date();

            const dateStr = now.toLocaleDateString('th-TH', { day: 'numeric', month: 'long', year: 'numeric' });

            document.getElementById('todayDateDisplay').innerText = dateStr;


            // คำนวณหาจันทร์-ศุกร์ ของสัปดาห์นี้

            const dayOfWeek = now.getDay(); // 0 (อาทิตย์) - 6 (เสาร์)

            const diffToMon = dayOfWeek === 0 ? -6 : 1 - dayOfWeek;

            const monday = new Date(now);

            monday.setDate(now.getDate() + diffToMon);

            

            const friday = new Date(monday);

            friday.setDate(monday.getDate() + 4);


            const monStr = monday.toLocaleDateString('th-TH', { day: 'numeric', month: 'short' });

            const friStr = friday.toLocaleDateString('th-TH', { day: 'numeric', month: 'short', year: 'numeric' });

            

            document.getElementById('weekRangeDisplay').innerText = `สัปดาห์นี้: ${monStr} - ${friStr}`;

        }


        function formatTimeDisplay(val) {

            if (!val) return "--:--";

            if (typeof val === 'string' && (val.includes("T") || val.includes("1899"))) {

                const d = new Date(val);

                const h = String(d.getHours()).padStart(2, '0');

                const m = String(d.getMinutes()).padStart(2, '0');

                return `${h}:${m}`;

            }

            return val;

        }


        async function loadData() {

            updateDateHeader();

            try {

                const res = await fetch(`${WEB_APP_URL}?action=getTasks`);

                state.tasks = await res.json();

                render();

            } catch (err) { console.error("Error:", err); }

        }


        function render() {

            const container = document.getElementById('tasksContainer');

            container.innerHTML = "";

            state.tasks.filter(t => t.day === state.currentDay).sort((a,b) => a.startTime.localeCompare(b.startTime)).forEach(task => {

                const card = document.createElement('div');

                card.className = "border rounded-xl p-4 flex justify-between items-center bg-white shadow-sm hover:border-indigo-300";

                card.innerHTML = `

                    <div class="flex-1">

                        <h4 class="font-bold text-lg text-slate-800">${task.name}</h4>

                        <div class="text-sm text-slate-600 mt-1">

                            <span class="font-medium text-indigo-700">${formatTimeDisplay(task.startTime)} - ${formatTimeDisplay(task.endTime)}</span> 

                            | <span>${task.location}</span>

                            | <span class="text-emerald-600 font-semibold"><i class="fa-solid fa-user mr-1"></i>${task.assignee}</span>

                        </div>

                    </div>

                    <div class="flex gap-2">

                        <button onclick="editTask('${task.id}')" class="text-blue-500 hover:text-blue-700 p-2"><i class="fa-solid fa-pencil"></i></button>

                        <button onclick="deleteTask('${task.id}')" class="text-red-500 hover:text-red-700 p-2"><i class="fa-solid fa-trash"></i></button>

                    </div>

                `;

                container.appendChild(card);

            });

        }


        function editTask(id) {

            const task = state.tasks.find(t => t.id === id);

            if (!task) return;

            document.getElementById('editId').value = task.id;

            document.getElementById('taskName').value = task.name;

            document.getElementById('taskDay').value = task.day;

            document.getElementById('taskStartTime').value = task.startTime;

            document.getElementById('taskEndTime').value = task.endTime;

            document.getElementById('taskLocation').value = task.location;

            document.getElementById('taskAssignee').value = task.assignee;

            document.getElementById('modalTitle').innerText = "แก้ไขงาน";

            document.getElementById('taskModal').classList.remove('hidden');

        }


        async function saveTask(e) {

            e.preventDefault();

            const id = document.getElementById('editId').value;

            const task = {

                id: id || 'task-' + Date.now(),

                name: document.getElementById('taskName').value,

                day: document.getElementById('taskDay').value,

                startTime: document.getElementById('taskStartTime').value,

                endTime: document.getElementById('taskEndTime').value,

                location: document.getElementById('taskLocation').value,

                assignee: document.getElementById('taskAssignee').value,

                status: 'Pending'

            };

            const action = id ? 'edit' : 'add';

            await fetch(WEB_APP_URL, { method: 'POST', body: JSON.stringify({ action, task }) });

            closeModal();

            loadData();

        }


        function switchDay(day, el) {

            state.currentDay = day;

            document.querySelectorAll('.day-btn').forEach(b => b.classList.remove('active'));

            if(el) el.classList.add('active');

            const map = {'Monday': 'จันทร์', 'Tuesday': 'อังคาร', 'Wednesday': 'พุธ', 'Thursday': 'พฤหัสบดี', 'Friday': 'ศุกร์'};

            document.getElementById('currentDayTitle').innerText = "ตารางงานวัน" + map[day];

            render();

        }


        async function deleteTask(id) {

            if(!confirm("ลบงานนี้ใช่ไหม?")) return;

            await fetch(WEB_APP_URL, { method: 'POST', body: JSON.stringify({ action: 'delete', task: { id } }) });

            loadData();

        }


        async function clearDayTasks() {

            if(!confirm("ยืนยันลบงานทั้งหมดของวันนี้?")) return;

            const tasksToDelete = state.tasks.filter(t => t.day === state.currentDay);

            for (const t of tasksToDelete) {

                await fetch(WEB_APP_URL, { method: 'POST', body: JSON.stringify({ action: 'delete', task: { id: t.id } }) });

            }

            loadData();

        }


        function openModal() { document.getElementById('modalTitle').innerText = "เพิ่มงานใหม่"; document.getElementById('taskForm').reset(); document.getElementById('editId').value = ""; document.getElementById('taskModal').classList.remove('hidden'); }

        function closeModal() { document.getElementById('taskModal').classList.add('hidden'); }

        window.onload = loadData;

    </script>
